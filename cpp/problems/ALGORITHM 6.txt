#include <iostream>
#include <cmath>
#include <string>
#include <algorithm>
using namespace std;


int ReadNumber(string message)
{
	int number;

		cout << message << endl;
		cin >> number;

	return number;
}

void ReversedNumber(int number)
{
	int remainder = 0;
	int sum = 0;
	while (number > 0)
	{
		remainder = number % 10;
		number = number / 10;
		sum += remainder;
	}
	
	cout << "The sum of your digits is: " << sum << endl;

	
}


int main()
{  
	int number = ReadNumber("Please enter your number to reverse it:");
    ReversedNumber(number);

	return 0;
}
