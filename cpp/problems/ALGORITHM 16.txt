void PrintNumberFromAAAtoZZZ()
{
	for (int i = 65; i <= 90; i++)
	{
		for (int j = 65; j <= 90; j++)
		{
			for (int m = 65; m <= 90; m++)
			{
				cout << char(i) << char(j) << char(m) << endl;
			}
		}
	}
}

{  
	PrintNumberFromAAAtoZZZ();

	return 0;
