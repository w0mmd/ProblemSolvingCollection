#include <iostream>
#include <cmath>
#include <string>
#include <algorithm>
#include <cstdlib>
using namespace std;

int RandomNumber(int from, int to)
{
	int RandNumber = rand() % (to - from + 1) + from;

	return RandNumber;
}

int main()
{  
	srand((unsigned)time(NULL));

		cout << RandomNumber(1,5);

	return 0;
}
