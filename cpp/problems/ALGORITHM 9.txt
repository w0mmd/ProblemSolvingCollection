int ReadPositiveNumber(string message)
{

}


int CheckFrequency(int number, int digittocheck)
{
	int digit = 0;
	int i = 0;
	while (number > 0)
	{

			digit = number % 10;
			if (digit == digittocheck)
			{
				i++;
			}
			number = number / 10;

	}
	return i;
}

void CheckFrequencyAllNumbers(int number)
{
	int num = 0;
	for (int i = 0; i < 10; i++)
	{
		num = CheckFrequency(number, i);

		if (num > 0)
		{ 
			cout << "The number " << i << " is repeated " << num << " times.\n";
		}
	}
}
