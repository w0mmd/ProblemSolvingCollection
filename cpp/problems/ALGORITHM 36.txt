bool AskToContinue()
{
	int number2 = 0;
	
		cout << "Do you want to add more number? [0] if No, [1] if Yes?" << endl;
		cin >> number2;

		return (number2 == 1);
	
}

int ReadPositiveNumberInArray(string message, int arr[100])
{
	int number = 0;
	int counter = 0;
	do {
		cout << message << endl;
		cin >> number;

		if (number > 0)
		{
			arr[counter] = number;
			counter++;
		}
		else {
			cout << "Please only positive numbers to enter:" << endl;
		}

	} while (AskToContinue() && counter < 100);


	return counter;
}

void PrintArray( int arr[100], int length)
{
	cout << "Array Length: " << length << endl;
	cout << "Array Elements: ";

	for (int i = 0; i < length; i++)
	{
		cout << arr[i] << " ";
	}
}

int main()
{  
	int arr[100];

	int length = ReadPositiveNumberInArray("Please enter your number:", arr);

	PrintArray(arr, length);
	return 0;
}

#include <iostream>

int ReadNumber()
{
	int number = 0;

	cout << "Please enter your number?" << endl;
	cin >> number;

	return number;
}

void AddArrayElement(int number, int arr[100], int &arrlength)
{
	arrlength++;
	arr[arrlength - 1] = number;
}

void EnterUserNumberInArray(int arr[100], int &arrlength)
{
	bool AddMore = true;

	do {

		AddArrayElement(ReadNumber(), arr, arrlength);

		cout << "Do you want to add more number? [0] No, [1] Yes?\n";
		cin >> AddMore;

	} while (AddMore);
}

void PrintArray(int arr[100], int arrlength)
{
	cout << "Array length: " << arrlength << endl;
	cout << "Array elements: ";
	for (int i = 0; i < arrlength; i++)
	{
		cout << arr[i] << " ";
	}
}

int main()
{  
	int arr[100], arrlength = 0;
	
	EnterUserNumberInArray(arr, arrlength);

	PrintArray(arr, arrlength);
