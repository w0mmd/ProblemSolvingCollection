#include <iostream>
#include <cmath>
#include <string>
#include <algorithm>
using namespace std;

int ReadPositiveNumber(string message)
{
	int number = 0;

	do {
		cout << message << endl;
		cin >> number;
	} while (number <= 0);

	return number;
}

void PrintNumber(int number)
{
	for (int i = number; i >= 1; i--)
	{
		for (int j = 1; j <= i ; j++)
		{
			cout << number;
		}
		cout << endl;
		number = number - 1;
	}
}

int main()
{  
	PrintNumber(ReadPositiveNumber("Please enter a number to check if it is a Palindrome number or not:"));

	return 0;
}
