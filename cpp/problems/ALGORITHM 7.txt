#include <iostream>
#include <cmath>
#include <string>
#include <algorithm>
using namespace std;


int ReadNumber(string message)
{
	int number;

		cout << message << endl;
		cin >> number;

	return number;
}

int ReversedNumber(int number)
{
	int remainder = 0, number2 = 0;
	while (number > 0)
	{
		remainder = number % 10;
		number = number / 10;
		number2 = number2 * 10 + remainder;
	}
	
	return number2;
}


int main()
{  
	int number = ReadNumber("Please enter your number to reverse it:");
    cout << ReversedNumber(number);

	return 0;
}
