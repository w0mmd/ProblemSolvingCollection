#include <iostream>
#include <cmath>
#include <string>
#include <algorithm>
#include <cstdlib>
using namespace std;


float ReadNumber(string message)
{
	float number = 0;

		cout << message << endl;
		cin >> number;

	return number;
}

int FloorFunction(float number)
{
	int intNumber = (int)number;

	if (number < 0)
	{
		return number - 1;
	}
	else {
		return intNumber;
	}
		
}

int main()
{  
	float number = ReadNumber("Please enter your number:");

	cout << "My floor function: " << FloorFunction(number) << endl;
	cout << "C++ function: " << floor(number);

	return 0;
}
