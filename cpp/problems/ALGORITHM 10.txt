int reversednumber(int number)
{
	int remainder = 0, num2 = 0;

	while (number > 0)
	{
		remainder = number % 10;
		number = number / 10;
		num2 = num2 * 10 + remainder;
	}
	return num2;
}

void printdigits(int number)
{
	int remainder = 0;

	while (number > 0)
	{
		remainder = number % 10;
		number = number / 10;
		cout << remainder << endl;
	}
}
