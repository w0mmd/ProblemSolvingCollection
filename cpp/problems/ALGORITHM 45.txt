#include <iostream>
#include <cmath>
#include <string>
#include <algorithm>
#include <cstdlib>
using namespace std;


int ReadPositiveNumber(string message)
{
	int number = 0;

	do {
		cout << message << endl;
		cin >> number;
	} while (number <= 0);

	return number;
}

int RandomNumber(int from, int to)
{
	int RandNumber = rand() % (to - from + 1) + from;

	return RandNumber;
}

void FillArray(int arr[], int length)
{
	cout << "Array elements: \n";
	for (int i = 0; i < length; i++)
	{
		arr[i] = RandomNumber(-100, 100);
		cout << arr[i] << " ";
	}
}

void CountNegativeNumbers(int arr[], int length)
{
	int count = 0;
	for (int i = 0; i < length; i++)
	{
		if (arr[i] < 0)
		{
			count++;
		}
	}
	cout << "Positive numbers count is: " << count << endl;
}

int main()
{  
	srand((unsigned)time(NULL));

	int arr1[100];

	int number = ReadPositiveNumber("Please enter your number to start an array:");

	FillArray(arr1, number);
	cout << endl;
	CountNegativeNumbers(arr1, number);

	return 0;
}
