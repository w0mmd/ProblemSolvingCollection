#include <iostream>
#include <cmath>
#include <string>
#include <algorithm>
using namespace std;


void printheader()
{
	cout << "\t \t \t \t Multiplication Table From 1 to 10 \t \t \t \t \n" << endl;
	for (int i = 1; i <= 10; i++)
	{
		cout  << "\t" << i;
	}

	cout << "\n_____________________________________________________________________________________\n";
}

string tableseparator(int i)
{
	if (i >= 10)
	return "  |";

   else
	return "   |";
}

void printtable()
{
	printheader();
	for (int i = 1; i <= 10; i++)
	{
		cout << " " << i << tableseparator(i) << "\t";
		for (int j = 1; j <= 10; j++)
		{
			cout << i * j << "\t";
		}
		cout << endl;
	}
}


int main()
{  
	printtable();
	
	return 0;
}
