using namespace std;


string ReadText()
{
    string Message;

    cout << "Please enter your text:\n";
    getline(cin, Message);

    return Message;
}

string LowerAllString (string Text)
{

    for (short i = 0; i < Text.length(); i++)
    {
       
           Text[i] = tolower(Text[i]);
       
    }
    return Text;
 }

string UpperAllString (string Text)
{

    for (short i = 0; i < Text.length(); i++)
    {
       
           Text[i] = toupper(Text[i]);
       
    }
    return Text;
 }

int main()
{
    string Text = ReadText();

    cout << "String after conversion to small characters:\n";
   Text =  LowerAllString(Text);
   cout << Text << endl;

    cout << "\nString after conversion to capital characters:\n";
   Text =  UpperAllString(Text);
   cout << Text << endl;

    return 0;  
}
